Name

JPClient

Description

An RPC Client for JobPractical

Synopsis

    use JPClient;
    my $jc = JPClient->new( apikey => 'key' );
    $jc->account->login({
        email => foo@bar.org,
        @morestuff,
        persist_login => 1,
    });
    $jc->info->whatever;
    $jc->account->logout();
